    <script>
        // --- State Management ---
        let totalScanned = 0;
        let totalDiseased = 0;
        let severitySum = 0;

        // --- DOM Elements ---
        const dropZone = document.getElementById('dropZone');
        const fileInput = document.getElementById('fileInput');
        const resultsContainer = document.getElementById('results');
        const loader = document.getElementById('loader');
        const historyModal = document.getElementById('historyModal');
        const historyList = document.getElementById('historyList');

        // --- Event Listeners ---
        
        // Drag & Drop
        ['dragenter', 'dragover', 'dragleave', 'drop'].forEach(eventName => {
            dropZone.addEventListener(eventName, preventDefaults, false);
        });

        function preventDefaults(e) {
            e.preventDefault();
            e.stopPropagation();
        }

        ['dragenter', 'dragover'].forEach(eventName => {
            dropZone.addEventListener(eventName, () => dropZone.classList.add('drag-active'), false);
        });

        ['dragleave', 'drop'].forEach(eventName => {
            dropZone.addEventListener(eventName, () => dropZone.classList.remove('drag-active'), false);
        });

        dropZone.addEventListener('drop', handleDrop, false);
        fileInput.addEventListener('change', handleFiles, false);

        // --- Core Logic ---

        function handleDrop(e) {
            const dt = e.dataTransfer;
            const files = dt.files;
            uploadFiles(files);
        }

        function handleFiles() {
            const files = this.files;
            uploadFiles(files);
        }

        async function uploadFiles(files) {
            if (files.length === 0) return;

            const formData = new FormData();
            for (let i = 0; i < files.length; i++) {
                formData.append('images', files[i]);
            }

            // Show Loader
            loader.style.display = 'flex';

            try {
                const response = await fetch('/predict', {
                    method: 'POST',
                    body: formData
                });

                if (!response.ok) throw new Error('Server error');

                const data = await response.json();
                
                if (data.success) {
                    loader.style.display = 'none';
                    processResults(data.results);
                    showToast('Analysis complete!', 'success');
                    // Reset input
                    fileInput.value = '';
                } else {
                    throw new Error(data.error || 'Unknown error');
                }

            } catch (error) {
                loader.style.display = 'none';
                showToast(error.message, 'error');
                console.error(error);
            }
        }

        function processResults(results) {
            results.forEach(res => {
                // Update Stats
                totalScanned++;
                if (res.disease_percentage > 10) totalDiseased++; // Assume >10% is diseased
                severitySum += res.disease_percentage;
                updateDashboard();

                // Render Card
                const card = document.createElement('div');
                card.className = 'result-card';
                
                // Determine tags
                const isHealthy = res.disease_percentage < 15; // Threshold
                const healthClass = isHealthy ? 'healthy' : '';
                const healthLabel = isHealthy ? 'Healthy' : 'Diseased';
                
                // URL Paths
                const originalUrl = `static/uploads/${res.filename}`;
                const resultUrl = `static/uploads/${res.image_result}`;

                card.innerHTML = `
                    <div class="comparison-view" id="comp-${res.id}">
                        <span class="badge badge-original">Original</span>
                        <span class="badge badge-segment">Segmented</span>
                        
                        <!-- Background Image (Segmented/Result) -->
                        <img src="${resultUrl}" alt="Segmented" style="width:100%; height:100%; object-fit:contain;">
                        
                        <!-- Foreground Image (Original) - Clipped -->
                        <div class="img-overlay">
                            <img src="${originalUrl}" alt="Original" style="width: 100%; height:100%; max-width:none; object-fit:contain;">
                        </div>
                        
                        <!-- Slider Handle -->
                        <div class="slider-handle">
                            <i class="fa-solid fa-arrows-left-right"></i>
                        </div>
                    </div>
                    
                    <div class="card-body">
                        <div class="prediction-header">
                            <h3>${res.prediction}</h3>
                            <span class="disease-tag ${healthClass}">${healthLabel}</span>
                        </div>
                        
                        <div class="metrics-grid">
                            <div class="metric-item">
                                <label>Confidence</label>
                                <div class="metric-value">${res.confidence}%</div>
                                <div class="progress-bg">
                                    <div class="progress-fill" style="width: ${res.confidence}%; background: var(--secondary);"></div>
                                </div>
                            </div>
                            <div class="metric-item">
                                <label>Infected Area</label>
                                <div class="metric-value">${res.disease_percentage}%</div>
                                <div class="progress-bg">
                                    <div class="progress-fill" style="width: ${res.disease_percentage}%; background: var(--danger);"></div>
                                </div>
                            </div>
                        </div>
                        
                        <div style="margin-top: 1rem; text-align: right;">
                            <a href="${resultUrl}" download="analysis_${res.filename}" style="color: var(--primary); text-decoration: none; font-size: 0.9rem;">
                                <i class="fa-solid fa-download"></i> Download Result
                            </a>
                        </div>
                    </div>
                `;

                resultsContainer.prepend(card);

                // Initialize Slider Logic for this card
                initSlider(`#comp-${res.id}`);
            });
        }

        // --- Slider Logic ---
        function initSlider(selector) {
            const container = document.querySelector(selector);
            if (!container) return;

            const overlay = container.querySelector('.img-overlay');
            const imgOverlay = overlay.querySelector('img');
            const imgBg = container.querySelector('img:first-of-type');
            const handle = container.querySelector('.slider-handle');

            let isDragging = false;

            const updateSlider = (x) => {
                const rect = container.getBoundingClientRect();
                let pos = x - rect.left;
                
                // Bounds
                if (pos < 0) pos = 0;
                if (pos > rect.width) pos = rect.width;
                
                const percentage = (pos / rect.width) * 100;
                
                overlay.style.width = `${percentage}%`;
                handle.style.left = `${percentage}%`;
                
                // Crucial: Sync the image inside the overlay so it doesn't squish
                // We use negative margin or width trickery usually, but 
                // with object-fit:contain and full width, it's tricky.
                // Simplified approach: Ensure image in overlay matches BG visual position
                imgOverlay.style.width = `${rect.width}px`;
                imgOverlay.style.height = `${rect.height}px`;
            };

            // Mouse Events
            container.addEventListener('mousedown', () => isDragging = true);
            window.addEventListener('mouseup', () => isDragging = false);
            container.addEventListener('mousemove', (e) => {
                if (!isDragging) return;
                updateSlider(e.clientX);
            });

            // Touch Events
            container.addEventListener('touchstart', () => isDragging = true);
            window.addEventListener('touchend', () => isDragging = false);
            container.addEventListener('touchmove', (e) => {
                if (!isDragging) return;
                updateSlider(e.touches[0].clientX);
            });

            // Initial set (resize observer handles resizing later)
            updateSlider(container.getBoundingClientRect().width / 2 + container.getBoundingClientRect().left);
        }

        // --- Dashboard & History ---
        function updateDashboard() {
            document.getElementById('statTotal').innerText = totalScanned;
            document.getElementById('statDiseased').innerText = totalDiseased;
            const avg = totalScanned > 0 ? (severitySum / totalScanned).toFixed(1) : 0;
            document.getElementById('statSeverity').innerText = `${avg}%`;
        }

        function toggleHistory() {
            historyModal.classList.toggle('open');
            if (historyModal.classList.contains('open')) {
                fetchHistory();
            }
        }

        async function fetchHistory() {
            try {
                const res = await fetch('/history');
                const data = await res.json();
                renderHistory(data.history);
            } catch (e) {
                console.error(e);
            }
        }

        function renderHistory(history) {
            historyList.innerHTML = '';
            if (!history || history.length === 0) {
                historyList.innerHTML = '<p style="color:#9ca3af; text-align:center;">No history found.</p>';
                return;
            }

            history.forEach(item => {
                const el = document.createElement('div');
                el.className = 'history-item';
                el.innerHTML = `
                    <img src="static/uploads/${item.result_image}" class="history-thumb" alt="thumb">
                    <div class="history-info">
                        <div style="font-weight:600; font-size:0.9rem;">${item.prediction}</div>
                        <div style="font-size:0.8rem; color:var(--primary);">${item.confidence}% Confident</div>
                        <div class="history-date">${item.timestamp}</div>
                    </div>
                `;
                historyList.appendChild(el);
            });
        }

        // --- Utilities ---
        function showToast(message, type = 'success') {
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;
            toast.innerHTML = `
                <i class="fa-solid ${type === 'success' ? 'fa-circle-check' : 'fa-circle-exclamation'}"></i>
                <span>${message}</span>
            `;
            document.getElementById('toastContainer').appendChild(toast);
            
            setTimeout(() => {
                toast.style.opacity = '0';
                setTimeout(() => toast.remove(), 300);
            }, 3000);
        }

        // Initial Load (Optional: fetch history on load to update stats if persisted)
        window.onload = () => {
            // Optional: fetch('/history')... to populate stats from previous sessions
        };

    </script>
